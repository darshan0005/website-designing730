import pygame import sys # Initialize Pygame pygame.init() # Set up display WIDTH, HEIGHT = 800, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Car Simulation") # Load car image car_img = pygame.image.load("car.png") # You need a car.png image in same folder car_img = pygame.transform.scale(car_img, (60, 100)) # Resize car # Initial position and speed x, y = WIDTH // 2, HEIGHT // 2 speed = 5 # Clock clock = pygame.time.Clock() # Game loop running = True while running: screen.fill((50, 150, 50)) # Green background (like a road) for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # Movement keys = pygame.key.get_pressed() if keys[pygame.K_LEFT]: x -= speed if keys[pygame.K_RIGHT]: x += speed if keys[pygame.K_UP]: y -= speed if keys[pygame.K_DOWN]: y += speed # Draw car screen.blit(car_img, (x, y)) pygame.display.update() clock.tick(60) pygame.quit() sys.exit()
